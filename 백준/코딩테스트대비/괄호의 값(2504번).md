import sys

def isVPS(ps):
    stack = []
    for c in ps:
        if c == '(' or c == '[':
            stack.append(c)
        elif c == ')' and stack:
            if stack[-1] == '(':
                stack = stack[:-1]
            else:
                stack.append(c)
        elif c == ']' and stack:
            if stack[-1] == '[':
                stack = stack[:-1]
            else:
                stack.append(c)
        else:
            stack.append(c)
    if stack:
        return False
    else:
        return True

def calculateVPS(ps):
    stack = []
    for c in ps:
        if c == '(' or c == '[':
            stack.append(c)
        elif c == ')':
            if stack[-1] == '(':
                stack[-1] = 2
            else:
                tmp = 0
                for i in range(len(stack) - 1, -1, -1):
                    if stack[i] == '(':
                        stack[-1] = tmp * 2
                        break
                    else:
                        tmp += stack[i]
                        stack = stack[:-1]
        elif c == ']':
            if stack[-1] == '[':
                stack[-1] = 3
            else:
                tmp = 0
                for i in range(len(stack) - 1, -1, -1):
                    if stack[i] == '[':
                        stack[-1] = tmp * 3
                        break
                    else:
                        tmp += stack[i]
                        stack = stack[:-1]

    return sum(stack)

ps = sys.stdin.readline().strip()
if isVPS(ps):
    print(calculateVPS(ps))
else:
    print(0)
