문제 링크 : https://www.acmicpc.net/problem/2493

```
# 시간 초과 코드
import sys

n = int(sys.stdin.readline())
towers = list(map(int, sys.stdin.readline().split()))
answers = [0]

for i in range(1, n):
    for j in range(i-1, -1, -1):
        if towers[j] > towers[i] :
            answers.append(j+1)
            break
    else:
        answers.append(0)
        
print(" ".join(map(str, answers)))
```
