   def dfs(graph, start_node):
       visited = []
       stack = [start_node]
   
       while stack:
           current = stack.pop()
           if current not in visited:
               visited.append(current)
               if current in graph:
                   temp = graph[current]
                   temp.sort(reverse=True)
                   stack.extend(temp)
       return ' '.join(map(str,visited))
       
   def bfs(graph, start_node):
       visited = []
       queue = [start_node]
   
       while queue:
           current = queue.pop(0)
           if current not in visited:
               visited.append(current)
               if current in graph:
                   temp = graph[current]
                   temp.sort()
                   queue.extend(temp)
   
       return ' '.join(map(str,visited))
   
   n,m,v = map(int, input().split())
   graph = {}
   for i in range(m):
       a,b = map(int, input().split())
       graph[a] = graph.get(a, []) + [b]
       graph[b] = graph.get(b, []) + [a]
   print(dfs(graph, v))
   print(bfs(graph, v))
