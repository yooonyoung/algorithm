1. N과 M (1) https://www.acmicpc.net/problem/15649

   ```
   n,m = map(int, input().split())
   visited = [False] * n
   result = []
   
   def solve(depth, n, m):
       if depth == m:
           print(' '.join(map(str, result)))
           return
   
       for i in range(len(visited)):
           if not visited[i]:
               visited[i] = True
               result.append(i+1)
               solve(depth+1, n, m)
               visited[i] = False
               result.pop()
   solve(0, n, m)
   ```

2. N과 M (2) https://www.acmicpc.net/problem/15650

   ```
   n,m = map(int, input().split())
   visited = [False] * n
   result = []
   
   def solve(depth, index, n, m):
       if depth == m:
           print(' '.join(map(str, result)))
           return
   
       for i in range(index, len(visited)):
           if not visited[i]:
               visited[i] = True
               result.append(i+1)
               solve(depth+1, i+1, n, m)
               visited[i] = False
               result.pop()
   solve(0, 0, n, m)
   ```
